app.directive('myfiles', function ($http) {
            return {
                restrict: 'E',
                link: function (scope, element, attrs) {
                    var html = '[ul]';
                    var fs = require('fs');
                    files = fs.readdirSync('./slides');
                    for (var c in files) {
                        html += '[li]' + files[c] + '[/li]';
                    }
                    html += '[/ul]';
                    element.html(html);
                }
            }
        });
