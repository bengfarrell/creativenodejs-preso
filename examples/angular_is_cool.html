app.directive('myfiles', function ($http) {
            return {
                restrict: 'E',
                link: function (scope, element, attrs) {
                    $http({method: 'GET', url: 'http://play.blastanova.com/json/webassets.json'}).then(function (result) {
                        var html = '[ul]'
                        for (var c in result.data.sources[0].assets) {
                            html += '[li]' + result.data.sources[0].assets[c].filename + '[/li]';
                        }
                        html += '[/ul]';
                         element.html(html);
                    });
                }
            }
        });
